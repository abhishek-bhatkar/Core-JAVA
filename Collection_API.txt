			ObjectOutputStream oos=new ObjectOutputStream(fos);
			oos.writeObject(mylist);
			oos.close();
			
			FileInputStream fis=new FileInputStream("F:\\ab1.txt");
			ObjectInputStream ois=new ObjectInputStream(fis);
			List<?> ref=(List<?>) ois.readObject();
			
			Iterator<?> itr=ref.iterator();
			while(itr.hasNext())
			{
				System.out.println(itr.next());
			}
			
		}
		catch(Exception ee)
		{
			ee.printStackTrace();
		}
	}

package collection3;
import java.util.*;

interface Game{
	void play();
}

class FootBall implements Game{
	public void play() {
		System.out.println("football");
	}
}

class Cricket implements Game{
	public void play() {
		System.out.println("Cricket");
	}
}

class Hokey implements Game{
	public void play() {
		System.out.println("Hokey");
	}
}


class Perform<T extends Game>{
	Perform(T t){
		t.play();
	}
}

public class program3 {

	public static void main(String[] args) {
		
		List<Game> data = new ArrayList<Game>();
		data.add(new Cricket());
		data.add(new Hokey());
		data.add(new FootBall());
	
		
		Iterator itr = data.iterator();
		while(itr.hasNext()) {
			new Perform((Game) itr.next());
		}

	}

}

class MyClass {
	private int prnNo;
	private String name;
	
	public MyClass(int prnNo, String name) {
		this.prnNo = prnNo;
		this.name = name;
	}
	public int getPrnNo() {
		return prnNo;
	}
	public void setPrnNo(int prnNo) {
		this.prnNo = prnNo;
	}
	public String getName() {
		return name;
	}
	public void setName(String name) {
		this.name = name;
	}
	
}

public class assignment2{
	
	public static void iterateData(Map<String, MyClass> data) {
		Set< Entry<String, MyClass> > d = data.entrySet();
		Iterator< Entry<String, MyClass> > itr = d.iterator();
		
		while( itr.hasNext() ) {
			MyClass temp = itr.next().getValue();
			System.out.println( temp.getPrnNo() );
			System.out.println( temp.getName() );
		}
	}
	
	public static void main(String[] args) {
	
		MyClass m1 = new MyClass(1, "a");
		MyClass m2 = new MyClass(2, "b");
		
		Map<String, MyClass> data = new HashMap<String, MyClass>();
		data.put("first",  m1);
		data.put("second", m2);
		
		iterateData(data);
		

	}

}
